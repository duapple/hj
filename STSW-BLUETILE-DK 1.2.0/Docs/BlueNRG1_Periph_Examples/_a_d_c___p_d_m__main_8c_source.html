410 void TC_IT_Callback(void)
411 {
412  uint16_t nBytes, nPCMs = 0;
413 
414  for ( nBytes = 0; nBytes< (FS/1000)*2; nBytes++) /* FS/1000 PCM samples(16 bit) -> (FS/1000)*2 in Bytes */
415  {
416  if((nBytes%2)==0) {
417  HP_Filter.Z = PCM_Buffer[nPCMs + PCM_BUFFER_SIZE/2] * AudioVolume;
418  HP_Filter.oldOut = (0xFC * (HP_Filter.oldOut + HP_Filter.Z - HP_Filter.oldIn)) / 256;
419  HP_Filter.oldIn = HP_Filter.Z;
420  PCM_Buffer[nPCMs + PCM_BUFFER_SIZE/2] = SaturaLH(HP_Filter.oldOut, -32760, 32760);
421  nPCMs++;
422  }
423  /* Send the PDM data through UART */
424  UART_SendData( ((uint8_t *)PCM_Buffer)[nBytes + (FS/1000)*2]); /* FS/1000 PCM samples(16 bit) -> (FS/1000)*2 in Bytes */
425  while (UART_GetFlagStatus(UART_FLAG_TXFE) == RESET);
426  }
427 
428 }
