267 void MFT_Configuration()
268 {
269  MFT_InitType timer_init;
270  NVIC_InitType NVIC_InitStructure;
271 
272  SysCtrl_PeripheralClockCmd(CLOCK_PERIPH_MTFX1, ENABLE);
273 
274  /* Init MFT1 in mode 2 */
275  timer_init.MFT_Clock1 = MFT_PRESCALED_CLK;
276  timer_init.MFT_Clock2 = MFT_NO_CLK;
277  timer_init.MFT_Mode = MFT_MODE_2;
278  timer_init.MFT_CRA = 0xFFFF;
279  timer_init.MFT_CRB = 0xFFFF;
280 
281  /* If a square waveform is @ input with a frequency of 10kHz => T = 100 us
282  *
283  * with Prescaler = 16 - 1; => 1us sampling period
284  * the counter number within a period is 100.
285  *
286  * with Prescaler = 1 - 1; => 62.5ns sampling period
287  * the counter number within a period is 16000.
288  */
289  timer_init.MFT_Prescaler = 1-1;
290 
291  MFT_Init(MFT1, &timer_init);
292 
293  MFT_SetCounter(MFT1, 0xFFFF, 0xFFFF);
294 
295  /* Enable MFT Interrupts */
296  NVIC_InitStructure.NVIC_IRQChannel = MFT1A_IRQn;
297  NVIC_InitStructure.NVIC_IRQChannelPreemptionPriority = LOW_PRIORITY;
298  NVIC_InitStructure.NVIC_IRQChannelCmd = ENABLE;
299  NVIC_Init(&NVIC_InitStructure);
300 }
