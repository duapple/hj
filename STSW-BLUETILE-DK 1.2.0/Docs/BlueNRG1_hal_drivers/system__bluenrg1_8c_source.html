620 /* Wait the Low speed oscillator stabilization */
621 void Wait_LS_Stabilization(void)
622 {
623  uint8_t ls_state;
624  uint16_t number_iteration;
625  volatile uint32_t period1;
626  uint32_t period2, tick_result;
627 
628  number_iteration = 0;
629  ls_state = WAIT_FREQ_MIN_STATE;
630  CKGEN_BLE->CLK32K_COUNT = SLOW_COUNT;
631 
632  while ((ls_state != LS_STABILIZATION_ENDED) && (number_iteration <= MAX_NUMBER_ITERATION)) {
633 
634  /* Stabilization State machine */
635  switch(ls_state)
636  {
637  case WAIT_FREQ_MIN_STATE:
638  {
639  /* Wait until the LS freq is minor of 20 KHz */
640  CKGEN_BLE->CLK32K_IT = 1;
641  CKGEN_BLE->CLK32K_PERIOD = 0;
642  while(CKGEN_BLE->CLK32K_IT == 0);
643  period1 = CKGEN_BLE->CLK32K_PERIOD;
644  if (period1 <= MIN_CLOCK_PERIOD) {
645  ls_state = WAIT_FREQ_STABILIZATION_STATE;
646  }
647  }
648  break;
649  case WAIT_FREQ_STABILIZATION_STATE:
650  {
651  CKGEN_BLE->CLK32K_IT = 1;
652  CKGEN_BLE->CLK32K_PERIOD = 0;
653  while(CKGEN_BLE->CLK32K_IT == 0);
654  period2 = CKGEN_BLE->CLK32K_PERIOD;
655  if (period2 > period1)
656  tick_result = period2 - period1;
657  else
658  tick_result = period1 - period2;
659  if (tick_result <= (period1>>10)) {
660  ls_state = LS_STABILIZATION_ENDED;
661  }
662  period1 = period2;
663  }
664  break;
665  default:
666  /* Nothing to do */
667  break;
668  }
669  number_iteration++;
670  }
671 }

825  cold_start_config[11] |= (1<<2);

832  }
833 #else
834 #error "No definition for BOR Configuration"
835 #endif
836

838  BLUE_CTRL->RADIO_CONFIG = 0x10000U | (uint16_t)((uint32_t)cold_start_config & 0x0000FFFFU);
839  while ((BLUE_CTRL->RADIO_CONFIG & 0x10000) != 0);
840 
841  /* Configure RF FrontEnd */
842  if (((Device_partInfo.die_major<<4)|(Device_partInfo.die_cut)) >= WA_DEVICE_VERSION) {
843  Set_RF_FrontEnd();
844  }
845 
846  }

849  * measurement is done automatically each time the
850  * device enters in active2 state and the HS is ready.
851  * The interrupt signals that a measurement is done.
852  */
853  while(CKGEN_BLE->CLK32K_IT == 0);
854  CKGEN_BLE->CLK32K_IT = 1;
855  CKGEN_BLE->CLK32K_COUNT = 23; //Restore the window length for slow clock measurement.
856  CKGEN_BLE->CLK32K_PERIOD = 0;

860  if (waitLS_Ready) {
861  current_time = BLUE_CTRL->CURRENT_TIME;

863 #if (LS_SOURCE == LS_SOURCE_INTERNAL_RO)
864  /* Wait the LS stabilization */
865  Wait_LS_Stabilization();
866 #endif
867  }

871 #if (FORCE_CORE_TO_16MHZ == 1)
872  /* AHB up converter command register write*/

874 #else

878 #endif
