220 struct ActionPacket
221 {
222  uint8_t StateMachineNo ; /* This parameter indicates the state machine number for this action. From 0 to 7. */
223  uint8_t ActionTag; /* The configuration of the current action.
224  * Action Tag: PLL_TRIG, TXRX, TIMER_WAKEUP, INC_CHAN, TIMESTAMP_POSITION, RELATIVE */
225  uint32_t WakeupTime; /* Contains the wakeup time in microsecond if it is relative.
226  * It should not be more than 24 bits if it is absolute.
227  * It only applies if TIMER_WAKEUP flag is set in ActionTag. */
228  uint32_t ReceiveWindowLength; /* Sets RX window size in microsecond. Applicable only for RX actions. */
229  uint8_t *data; /* Pointer to the array with the data to send (header, length and data field), for TX.
230  * Pointer to the array where the data received are copied, for RX.
231  * In case of RX, the array must have the max size MAX_PACKET_LENGTH. */
232  uint32_t status; /* The Status Register with the information on the action. */
233  uint32_t timestamp_receive; /* This field contains the timestamp when a packet is received.
234  * Intended to be used in the dataRoutine() callback routine. RX only. */
235  int32_t rssi; /* The rssi of the packet was received with. RX only. */
236  BlueTransStruct trans_packet; /* This is a linked list, which is going to be used by Hardware.
237  * User does not need to do anything. */
238  ActionPacket *next_true; /* Pointer to next ActionPacket if condRoutine() returns TRUE */
239  ActionPacket *next_false; /* Pointer to next ActionPacket if condRoutine() returns FALSE */
240  uint8_t (*condRoutine)(ActionPacket*); /* User callback that decide the next ActionPacket to use.
241  * It is time critical. Routine must end within 45 us. */
242  uint8_t (*dataRoutine)(ActionPacket*, ActionPacket*); /* User callback for managing data. */
243  uint8_t trans_config; /* This is for configuring the device for TX or RX. User does not need to do anything. */
244 };
